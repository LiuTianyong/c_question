## 1.字符串排序

```c
#include <stdio.h>
#include "string.h"

void fun(char *str,int num);

int main() {

    char str[10] = "DBGEACF";
    char str2[10] = "CEAedca";

    int len = strlen(str);
    fun(str,len);

    printf("\n");
    len = strlen(str2);
    fun(str2,len);

    return 0;
}

void fun(char *str,int num) {
    char temp;
    int len = strlen(str);
    for (int i = 1; i < len - 2; ++i) {
        for (int j = 1; j < len - i - 1; ++j) {
            if (str[j] < str[j + 1]) {
                temp = str[j];
                str[j] = str[j + 1];
                str[j + 1] = temp;
            }
        }
    }
    for (int i = 0; i < strlen(str); ++i) {
        printf("%c ", str[i]);
    }
}
```

## 2.N以内素数

```c
#include <stdio.h>

#define MaxSize 100

int fun(int lim,int a[MaxSize]);

int main() {

    int b[MaxSize];

    int n = fun(10,b);

    return 0;
}

int fun(int lim,int a[MaxSize]){

    int numer = 0;
    int k = 0;
    for (int i = 1; i <= lim; ++i) {
        
        int m = (int)sqrt( (double)i );
        
        int flag = 1;
        for (int j = 2; j <= m; ++j) {
            if(i % j == 0){
                flag = 0;
                break;
            }
        }
        if(flag){
            a[k] = i;
            k++;
            numer++;
        }
    }

    for (int i = 0; i < numer; ++i) {
        printf("%d ",a[i]);
    }

    return numer;
}
```

## 3.字符串的前N个*

```text
假定输入的字符半小只包含字母和*号,清编写函数fun,它的功能是;使字符串的前导*号不得多于n个;若多于n个,则删除多余的*号;若少于或等于n个,则什么也不做,字符串中间和尾部的*号不删除。的数fun中给出的语句仅供参考。例如，字符串中的内容为中家球米嚷京术*BC*DEF*G**水*，若n的值为4;删除后，字符串中的内容应当是;本来事*A*BC*DEF*G**来*;若n的值为8,则字符串中的内容仍为:。**事*水**A*BC*DEF*G**中*。n的值在主函数中输入。
```



```c
#include <stdio.h>

#define N 50

void fun(char *str,int n);

int main() {
    char str[N] = "*******A*BC*DEF*G****";
    fun(str,4);

    return 0;
}
void fun(char *str,int n){

    int i = 0,j,k=0;
    while (str[k]=='*')k++;

    if(k > n){
        i = n;
        j = k;
        for (; str[j] != '\0' ; j++) {
            str[i++] = str[j];
            str[i] = 0;
        }
    }

    printf("%s",str);
}
```

## 4.N个字符串中找出最长的

##### **p 指针传参数问题

```c
#include <stdio.h>
#include "string.h"

#define NUM 5

void fun(char (*a)[81],int num,char **max);

int main() {

    char a[NUM][81] = {
            "dkhghgjgkjkjg55475647545764jkg",
            "fjfhfhfh",
            "djfjfjggjdl",
            "dj55555dl",
            "djf99999999999525114jdl"
    };

    char *max;
    fun(a,5,&max);
    return 0;
}

void fun(char (*a)[81],int num,char **max){
    int maxlen = strlen(a[0]);

    *max = a[0];
    for (int i = 0; i < num; ++i) {
        if(strlen(a[i]) >  maxlen){
            *max = a[i];
            maxlen = strlen(a[i]);
        }
    }

    printf("%s",*max);
}
```

## 5.删除前*

```c
#include <stdio.h>

void fun(char *a);

int main() {

    char a[] = "*******A*BC*DEF*G****";
    fun(a);
    return 0;
}

void fun(char *a){

    int k = 0;
    while (a[k++] == '*');			//注意k位置
	
    int i = 0;
    for (i = 0; a[i] != '\0'; ++i) {
        a[i] = a[i+k - 1];			// 注意-1
    }

    a[i] = '\0';
    printf("%s",a);
}
```

## 6.cos(x) - x=0

```
利用以下所示的简单迭代方法求方程：cos(x)-x=0的一个实根。
Xn+1=cos(Xn)
迭代步骤如下：
(1)取X1初值为0.0；
(2)X0=X1，把X1的值赋给X0；
(3)X1=cos(X0)，求出一个新的X1；
(4)若X0-X1的绝对值小于0.000001，执行步骤(5)，否则执行步骤(2)；
(5)所求X1就是方程cos(X)-X=0的一个实根，作为函数值返回。
```



```c
#include <stdio.h>
#include "math.h"


float fun();

int main() {

    float f = fun();

    printf("%f",f);
    return 0;
}

float fun()
{
    float x1=0.0,x0;
    while(1)
    {
        x0=x1;
        x1=cos(x0);
        if(fabs(x0-x1)<0.000001)
            break;
    }
    return x1;
}
```
